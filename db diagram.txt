// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs

Table users {
    id integer [primary key]
    username varchar
    role_id varchar
    created_at timestamp
}

Table role {
    id integer
    name varchar
    created_at timestamp
}

Table permission {
    id integer
    name varchar
    created_at timestamp
}

Table author {
    id integer
    name varchar
    created_at timestamp
}

Table publisher {
    id integer
    name varchar
    created_at timestamp
}

Table category {
    id integer
    name varchar
    created_at timestamp
}

Table role_permission {
    id integer [primary key]
    role_id integer
    permission_id integer
    created_at timestamp
}

Table books {
    id integer [primary key]
    title varchar
    author_id integer
    description text
    cover_image varchar
    publisher_id integer
    publication_date date
    category_id integer
    isbn integer
    page_count integer
    user_id integer [not null]
    status varchar
    created_at timestamp
}

Table customer_reviews {
    id integer [primary key]
    title varchar
    comment text
    rating_stars integer // [range(1,5)]
    user_id integer [not null]
    book_id integer [not null]
    created_at timestamp
}

Ref: role_permission.role_id > role.id
Ref: role_permission.permission_id > permission.id
Ref: users.role_id < role_permission.role_id

Ref: books.user_id < users.id
Ref: customer_reviews.user_id > users.id
Ref: customer_reviews.book_id > books.id

Ref: books.category_id > category.id
Ref: books.publisher_id > publisher.id
Ref: books.author_id > author.id
